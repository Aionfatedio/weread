import React, { useMemo, useState } from 'react';
import { ChevronDown, ChevronUp, BookOpen } from 'lucide-react';

const PROGRESS_BASE_PERCENT = 30;

const clampPercent = (value) => Math.min(100, Math.max(0, value));

const getRatioPercent = (value, base) => {
  if (!Number.isFinite(value) || !Number.isFinite(base) || base <= 0) return 0;
  return clampPercent((value / base) * 100);
};

const getWeightedProgressPercent = (ratioPercent, basePercent = PROGRESS_BASE_PERCENT) => {
  const safeBasePercent = clampPercent(basePercent);
  const safeRatioPercent = clampPercent(ratioPercent);

  // 展示宽度 = 固有可读宽度 + 真实比例在剩余空间中的权重增量。
  // 例如 base=25，ratio=50，则展示宽度为 25 + 50% * 75 = 62.5。
  return clampPercent(safeBasePercent + (safeRatioPercent / 100) * (100 - safeBasePercent));
};

const getMonthBarWidth = ({ isExpanded, isAnyExpanded, progress }) => {
  // 保留原有交互：当某个月展开时，其余未展开月份动态充满至 100%；
  // 全部收回后，再恢复到“基础可读宽度 + 真实权重增量”的比例。
  if (isAnyExpanded && !isExpanded) return 100;
  return clampPercent(progress);
};


// 将展示用时长统一转换为分钟数。
// 注意：0 就是 0，不再返回 1，避免“兜底值”参与真实比例计算。
const parseTimeToMinutes = (timeValue) => {
  if (typeof timeValue === 'number') return Math.max(0, timeValue);
  if (typeof timeValue !== 'string') return 0;

  const hourMatch = timeValue.match(/(\d+(?:\.\d+)?)\s*小时/);
  const minuteMatch = timeValue.match(/(\d+(?:\.\d+)?)\s*分钟/);

  const hours = hourMatch ? Number(hourMatch[1]) : 0;
  const minutes = minuteMatch ? Number(minuteMatch[1]) : 0;

  return Math.max(0, hours * 60 + minutes);
};

const buildMonthlyRecords = (monthlyRecords) => {
  const recordsWithMinutes = monthlyRecords.map((record) => ({
    ...record,
    totalMinutes: parseTimeToMinutes(record.time),
    days: record.days?.map((day) => ({
      ...day,
      totalMinutes: parseTimeToMinutes(day.time),
    })),
  }));

  const maxMonthMinutes = Math.max(
    0,
    ...recordsWithMinutes.map((record) => record.totalMinutes)
  );

  return recordsWithMinutes.map((record) => {
    const maxDayMinutes = Math.max(
      0,
      ...(record.days ?? []).map((day) => day.totalMinutes)
    );

    const monthRatioProgress = getRatioPercent(record.totalMinutes, maxMonthMinutes);

    return {
      ...record,
      // 月度条也遵循“基础可读宽度 + 真实权重增量”的展示机制。
      ratioProgress: monthRatioProgress,
      calculatedProgress: getWeightedProgressPercent(monthRatioProgress),
      days: record.days?.map((day) => {
        const ratioProgress = getRatioPercent(day.totalMinutes, maxDayMinutes);

        return {
          ...day,
          // 真实权重：保留原始比例，用于后续调试、提示或数据分析。
          ratioProgress,
          // 展示权重：在 25% 基础可读宽度上叠加真实比例增量。
          // 这样文字仍然在进度条内部右侧，并随条形宽度移动。
          calculatedProgress: getWeightedProgressPercent(ratioProgress),
        };
      }),
    };
  });
};

export default function App() {
  const [isOpen, setIsOpen] = useState(true);
  const [expandedMonth, setExpandedMonth] = useState(1); // 默认展开4月以便预览效果
  const [animateBars, setAnimateBars] = useState(true);

  // 模拟书籍数据
  const bookData = {
    title: "(河图文化) 平妖传-qsnfj2010.txt",
    coverUrl: "https://images.unsplash.com/photo-1544947950-fa07a98d237f?q=80&w=200&auto=format&fit=crop",
    totalWords: "45.6",
    totalDurationHours: 9,
    totalDurationMinutes: 6,
    startDate: "3月10日",
    readingDays: 15,
    lastRead: "今天",
    maxDailyHours: 5,
    maxDailyMinutes: 33,
    maxDailyDate: "3月12日",
    monthlyRecords: [
      { month: "5月", time: "25分钟" },
      { 
        month: "4月", time: "41分钟",
        days: [
          { date: "29日", time: "4分钟" },
          { date: "28日", time: "10分钟" },
          { date: "27日", time: "9分钟" },
          { date: "26日", time: "12分钟" },
          { date: "5日", time: "1分钟" }, 
          { date: "2日", time: "3分钟" },
        ]
      },
      { month: "3月", time: "8小时" },
    ]
  };

  const processedMonthlyRecords = useMemo(
    () => buildMonthlyRecords(bookData.monthlyRecords),
    [bookData.monthlyRecords]
  );

  const handleExpand = (index) => {
    if (expandedMonth === index) {
      setExpandedMonth(null);
      return;
    }

    setExpandedMonth(index);
    setAnimateBars(false);
    window.setTimeout(() => setAnimateBars(true), 50);
  };

  return (
    <div className="min-h-screen bg-[#121212] flex items-center justify-center p-4 font-sans text-white">
      {!isOpen && (
        <button 
          onClick={() => setIsOpen(true)}
          className="px-6 py-3 bg-blue-500 hover:bg-blue-600 rounded-full font-medium transition-colors flex items-center gap-2"
        >
          <BookOpen size={20} />
          打开书籍数据
        </button>
      )}

      {isOpen && (
        <div className="fixed inset-0 bg-black/60 backdrop-blur-sm flex items-center justify-center z-50 p-4 transition-opacity duration-300">
          
          <div className="bg-[#1e1e1e] w-full max-w-[380px] rounded-[24px] shadow-2xl overflow-hidden relative animate-in fade-in zoom-in-95 duration-200">
            
            <div className="p-6 pb-5">
              {/* 顶部：书籍基本信息 */}
              <div className="flex justify-between items-start mb-6 mt-2">
                <div className="flex-1 pr-6 flex flex-col justify-between">
                  <h2 className="text-[20px] font-bold text-gray-100 leading-snug tracking-wide line-clamp-2">
                    {bookData.title}
                  </h2>
                  <div className="mt-3 text-[13px] text-gray-400">
                    上次阅读 · {bookData.lastRead}
                  </div>
                </div>
                <div className="w-[72px] h-[100px] flex-shrink-0 rounded shadow-md overflow-hidden bg-white/5 border border-white/10">
                  <img 
                    src={bookData.coverUrl} 
                    alt="Book Cover" 
                    className="w-full h-full object-cover"
                  />
                </div>
              </div>

              {/* 中间：核心数据统计卡片 */}
              <div className="bg-[#2a2a2a] rounded-[16px] p-5 mb-4">
                <div className="flex justify-around mb-2 mt-1">
                  <div className="flex flex-col items-center justify-center px-1">
                    <span className="text-[13px] text-gray-400 mb-1">总字数</span>
                    <div className="flex items-baseline gap-[2px]">
                      <span className="text-[26px] font-semibold text-gray-100">{bookData.totalWords}</span>
                      <span className="text-[12px] text-gray-300 font-normal">万</span>
                    </div>
                  </div>

                  <div className="flex flex-col items-center justify-center px-1">
                    <span className="text-[13px] text-gray-400 mb-1">阅读天数</span>
                    <div className="flex items-baseline gap-[2px]">
                      <span className="text-[26px] font-semibold text-gray-100">{bookData.readingDays}</span>
                      <span className="text-[12px] text-gray-300 font-normal">天</span>
                    </div>
                  </div>
                </div>
              </div>

              {/* 下半部分：阅读记录明细 */}
              <div className="bg-[#2a2a2a] rounded-[16px] p-5">
                
                {/* 累计时长 与 单日阅读最久 */}
                <div className="flex justify-between mb-6 px-1">
                  <div className="flex-1">
                    <div className="text-[13px] text-gray-400 mb-1">累计时长</div>
                    <div className="flex items-baseline gap-[2px]">
                      <span className="text-[22px] font-bold text-gray-100">{bookData.totalDurationHours}</span>
                      <span className="text-[12px] text-gray-300 font-normal mr-1">小时</span>
                      <span className="text-[22px] font-bold text-gray-100">{bookData.totalDurationMinutes}</span>
                      <span className="text-[12px] text-gray-300 font-normal">分钟</span>
                    </div>
                    <div className="text-[11px] text-gray-500 mt-1">{bookData.startDate}开始阅读</div>
                  </div>

                  <div className="flex-1 pl-6">
                    <div className="text-[13px] text-gray-400 mb-1">单日阅读最久</div>
                    <div className="flex items-baseline gap-[2px]">
                      <span className="text-[22px] font-bold text-gray-100">{bookData.maxDailyHours}</span>
                      <span className="text-[12px] text-gray-300 font-normal mr-1">小时</span>
                      <span className="text-[22px] font-bold text-gray-100">{bookData.maxDailyMinutes}</span>
                      <span className="text-[12px] text-gray-300 font-normal">分钟</span>
                    </div>
                    <div className="text-[11px] text-gray-500 mt-1">{bookData.maxDailyDate}</div>
                  </div>
                </div>

                {/* 月度及每日进度条列表 */}
                <div className="space-y-2 mt-2">
                  {processedMonthlyRecords.map((record, index) => {
                    const isExpanded = expandedMonth === index;
                    const isAnyExpanded = expandedMonth !== null;
                    const monthProgress = record.calculatedProgress;
                    const barWidthPercent = getMonthBarWidth({
                      isExpanded,
                      isAnyExpanded,
                      progress: monthProgress,
                    });
                    const barWidth = `${barWidthPercent}%`;

                    return (
                      <div key={`${record.month}-${index}`} className="flex flex-col">
                        
                        {/* 月份头部区 */}
                        {isExpanded ? (
                          <div className="flex justify-between items-center px-1 mb-1 mt-1">
                            <div className="text-[14px] text-gray-300">
                              <span className="font-bold">{record.month}</span>
                              <span className="mx-1">·</span>
                              <span className="font-bold">{record.time}</span>
                            </div>
                            <button 
                              onClick={() => handleExpand(index)}
                              className="w-7 h-7 flex items-center justify-center bg-white/10 rounded-full hover:bg-white/20 transition-colors"
                            >
                              <ChevronUp size={16} className="text-gray-300" />
                            </button>
                          </div>
                        ) : (
                          <div 
                            onClick={() => handleExpand(index)}
                            className="relative w-full h-[28px] bg-[#282828] rounded-[6px] flex items-center px-3 cursor-pointer hover:opacity-90 transition-opacity"
                          >
                            <div 
                              className="absolute left-0 top-0 bottom-0 bg-gradient-to-r from-[#5bcdae] to-[#45b799] rounded-[6px] transition-[width] duration-300 ease-out"
                              style={{ width: barWidth }}
                            ></div>
                            
                            <div className="relative z-10 w-full flex justify-between items-center text-[14px] text-white text-shadow-sm">
                              <div>
                                <span className="font-bold">{record.month}</span>
                                <span className="mx-1">·</span>
                                <span className="font-bold">{record.time}</span>
                              </div>
                              <ChevronDown size={16} className="text-white/80" />
                            </div>
                          </div>
                        )}

                        {/* 展开的每日明细 */}
                        <div 
                          className={`grid transition-[grid-template-rows,opacity,margin] duration-300 ease-in-out ${
                            isExpanded ? "grid-rows-[1fr] opacity-100 mt-2 mb-1" : "grid-rows-[0fr] opacity-0 mt-0 mb-0"
                          }`}
                        >
                          <div className="overflow-hidden space-y-2">
                            {record.days && record.days.map((day, dIdx) => {
                              const dayBarWidth = isExpanded && animateBars ? `${day.calculatedProgress}%` : '0%';

                              return (
                                <div key={`${day.date}-${dIdx}`} className="relative w-full h-[28px] bg-[#1e2227] rounded-[6px]">
                                  <div 
                                    className="absolute left-0 top-0 bottom-0 bg-gradient-to-r from-[#68b5e6] to-[#4b9ee2] rounded-[6px] transition-[width] duration-[300ms] ease-out overflow-hidden"
                                    style={{ width: dayBarWidth }}
                                  >
                                    <div className="h-full w-full flex justify-between items-center px-3">
                                      <span className="text-[13px] font-medium text-white whitespace-nowrap">{day.date}</span>
                                      <span className="text-[13px] font-bold text-white whitespace-nowrap">{day.time}</span>
                                    </div>
                                  </div>
                                </div>
                              );
                            })}
                          </div>
                        </div>

                      </div>
                    );
                  })}
                </div>
              </div>

            </div>

            {/* 底部按钮区 */}
            <div className="flex gap-3 px-6 pb-6 pt-2">
              <button className="flex-1 bg-[#3a3a3a] hover:bg-[#4a4a4a] text-[#e0e0e0] py-[14px] rounded-[12px] font-medium text-[15px] transition-colors">
                导出用户数据
              </button>
              <button className="flex-1 bg-[#0095ff] hover:bg-[#0084eb] text-white py-[14px] rounded-[12px] font-medium text-[15px] transition-colors shadow-lg shadow-blue-500/20">
                分享此书
              </button>
            </div>

          </div>
        </div>
      )}
    </div>
  );
}
